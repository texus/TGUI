396  #define TGUI_SIGNAL_VALUE_DECLARATION(TypeName, Type) \
397  /**************************************************************************************************************************** \
398  * @brief Signal to which the user can subscribe to get callbacks from
399  ****************************************************************************************************************************/ \
400  class TGUI_API Signal##TypeName : public Signal \
401  { \
402  public: \
403  using Delegate##TypeName = std::function<void(Type)>; \
404  using Delegate##TypeName##Ex = std::function<void(std::shared_ptr<Widget>, const std::string&, Type)>; \
405  using Signal::connect; \
406  \
407  /************************************************************************************************************************ \
408  * @brief Constructor
409  ************************************************************************************************************************/ \
410  Signal##TypeName(std::string&& name) : \
411  Signal{std::move(name), 1} \
412  { \
413  } \
414  \
415  /************************************************************************************************************************ \
416  * @brief Connects a signal handler that will be called when this signal is emitted
417  *
418  * @param handler Callback function that is given a Vector2f as argument
419  *
420  * @return Unique id of the connection
421  ************************************************************************************************************************/ \
422  unsigned int connect(const Delegate##TypeName& handler); \
423  \
424  /************************************************************************************************************************ \
425  * @brief Connects a signal handler that will be called when this signal is emitted
426  *
427  * @param handler Callback function that is given a pointer to the widget, the name of the signal and a Vector2f as arguments
428  *
429  * @return Unique id of the connection
430  ************************************************************************************************************************/ \
431  unsigned int connect(const Delegate##TypeName##Ex& handler); \
432  \
433  /************************************************************************************************************************ \
434  * @internal
435  * @brief Call all connected signal handlers
436  ************************************************************************************************************************/ \
437  bool emit(const Widget* widget, Type param) \
438  { \
439  if (m_handlers.empty()) \
440  return false; \
441  \
442  internal_signal::parameters[1] = static_cast<const void*>(&param); \
443  return Signal::emit(widget); \
444  } \
445  \
446  private: \
447  /************************************************************************************************************************ \
448  * @internal
449  * @brief Checks whether the unbound parameters match with this signal
450  * @return The index in the parameters list where the parameters will be stored
451  ************************************************************************************************************************/ \
452  unsigned int validateTypes(std::initializer_list<std::type_index> unboundParameters) const override; \
453  };
