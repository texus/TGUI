52  Clipping(const sf::RenderTarget& target, const sf::RenderStates& states, sf::Vector2f topLeft, sf::Vector2f size)
53  {
54  const sf::View& view = target.getView();
55  sf::Vector2f bottomRight = topLeft + size;
56 
57  // Calculate the scale factor of the view
58  float scaleViewX = target.getSize().x / view.getSize().x;
59  float scaleViewY = target.getSize().y / view.getSize().y;
60 
61  // Get the global position
62  sf::Vector2f topLeftPosition = {((topLeft.x - view.getCenter().x + (view.getSize().x / 2.f)) * view.getViewport().width) + (view.getSize().x * view.getViewport().left),
63  ((topLeft.y - view.getCenter().y + (view.getSize().y / 2.f)) * view.getViewport().height) + (view.getSize().y * view.getViewport().top)};
64  sf::Vector2f bottomRightPosition = {(bottomRight.x - view.getCenter().x + (view.getSize().x / 2.f)) * view.getViewport().width + (view.getSize().x * view.getViewport().left),
65  (bottomRight.y - view.getCenter().y + (view.getSize().y / 2.f)) * view.getViewport().height + (view.getSize().y * view.getViewport().top)};
66 
67  topLeftPosition = states.transform.transformPoint(topLeftPosition);
68  bottomRightPosition = states.transform.transformPoint(bottomRightPosition);
69 
70  // Get the old clipping area
71  glGetIntegerv(GL_SCISSOR_BOX, m_scissor);
72 
73  // Calculate the clipping area
74  GLint scissorLeft = std::max(static_cast<GLint>(topLeftPosition.x * scaleViewX), m_scissor[0]);
75  GLint scissorTop = std::max(static_cast<GLint>(topLeftPosition.y * scaleViewY), static_cast<GLint>(target.getSize().y) - m_scissor[1] - m_scissor[3]);
76  GLint scissorRight = std::min(static_cast<GLint>(bottomRightPosition.x * scaleViewX), m_scissor[0] + m_scissor[2]);
77  GLint scissorBottom = std::min(static_cast<GLint>(bottomRightPosition.y * scaleViewY), static_cast<GLint>(target.getSize().y) - m_scissor[1]);
78 
79  // If the object outside the window then don't draw anything
80  if (scissorRight < scissorLeft)
81  scissorRight = scissorLeft;
82  else if (scissorBottom < scissorTop)
83  scissorTop = scissorBottom;
84 
85  // Set the clipping area
86  glScissor(scissorLeft, target.getSize().y - scissorBottom, scissorRight - scissorLeft, scissorBottom - scissorTop);
87  }

98  ~Clipping()

100  glScissor(m_scissor[0], m_scissor[1], m_scissor[2], m_scissor[3]);
101  };
