128  unsigned int connect(const Func& func, const BoundArgs&... args)
129  {
130  const auto id = ++m_lastSignalId;
131 #if defined(__cpp_if_constexpr) && (__cpp_if_constexpr >= 201606L)
132  if constexpr(sizeof...(BoundArgs) == 0)
133  m_handlers[id] = func;
134  else
135 #endif
136  {
137  m_handlers[id] = [=]{ invokeFunc(func, args...); };
138  }
139 
140  return id;
141  }

245  template <typename Func, typename... Args>
246  static void invokeFunc(Func&& func, Args&&... args)
247  {
248  std::invoke(std::forward<Func>(func), std::forward<Args>(args)...);
249  }
250 #else
251  // std::invoke only exists in c++17 so we use our own implementation to support c++14 compilers
252  template <typename Func, typename... Args, typename std::enable_if<std::is_member_pointer<typename std::decay<Func>::type>::value>::type* = nullptr>
253  static void invokeFunc(Func&& func, Args&&... args)
